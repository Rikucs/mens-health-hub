# Git commands to run after installing Git

# Navigate to your project directory
cd "C:\xampp\htdocs\men's-health-hub"

# Initialize git repository
git init

# Add all files to staging
git add .

# Create initial commit
git commit -m "Initial commit: Men's Health Hub farewell magazine"

# Add your GitHub repository as remote origin
# Replace 'yourusername' with your actual GitHub username
git remote add origin https://github.com/yourusername/mens-health-hub.git

# Push to GitHub
git branch -M main
git push -u origin main
